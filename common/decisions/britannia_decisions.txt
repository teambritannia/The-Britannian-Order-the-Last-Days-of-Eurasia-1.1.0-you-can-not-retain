demand_german_tribute = {
	is_shown = {
		exists = c:GBR
		ROOT = c:GBR
		exists = c:GER
		c:GER = {
			has_journal_entry = je_german_british_split
		}
	}

	possible = {
		exists = c:GBR
		ROOT = c:GBR
		exists = c:GER
		c:GER = {
			has_journal_entry = je_german_british_split
		}
		custom_tooltip = {
			text = tt_tribute_cooldown
			NOT = {
				has_variable = demand_german_tribute_cooldown
			}
		}
	}

	when_taken = {
		hidden_effect = {
			set_variable = {
				name = demand_german_tribute_cooldown
				value = yes
				days = 730
			}		
		}
		c:GER = {
			trigger_event = {
				id = german_british_split.2
				days = 7
			}
		}
	}

	ai_chance = {
		base = 100
	}
}

		exists = c:GBR
		c:GBR = {
			has_journal_entry = je_economic_recovery
		}
	}

		country_rank >= rank_value:major_power
		custom_tooltip = {
			text = not_current_britannia_small_loan

			}
		}

	when_taken = {
		add_modifier = {
			name = britannia_small_loan
			months = normal_modifier_time
		}
		c:GBR = {
		trigger_event = {
			id = german_british_split.2
			days = 7
		}
	}

	ai_chance = {
		base = 100
	}
}
