revitalize_bureaucracy_button = {
	name = "revitalize_bureaucracy_button"
	desc = "revitalize_bureaucracy_button_desc"

	visible = {
		always = yes
	}

	ai_chance = {
		base = 100
	}

	possible = {
		bureaucracy >= 0 
		has_modifier = political_turmoil
	}

	effect = {
		remove_modifier = political_turmoil
		add_modifier = {
			name = political_strife
			months = normal_modifier_time
			is_decaying = yes
		}
	}
}

literacy_drive_button = {
	name = "literacy_drive_button"
	desc = "literacy_drive_button_desc"

	visible = {
		always = yes
	}

	ai_chance = {
		base = 100
		modifier = {
			trigger = {
				taking_loans = yes
			}
			add = -40
		}
		modifier = {
			trigger = {
				approaching_bureaucracy_shortage
			}
			add = -30
		}
	}

	possible = {
		NOT = {
			has_variable = literacy_drive_var
		}
	}

	effect = {
		add_modifier = {
			name = literacy_drive_expenses
			months = normal_modifier_time
		}
		add_modifier = {
			name = literacy_drive_effects
			months = literacy_drive_effects
		}
	}
}
