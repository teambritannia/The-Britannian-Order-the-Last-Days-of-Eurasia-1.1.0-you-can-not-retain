ig_center_communist = {
	color = hsv{ 0 0.56 0.58 }
	texture = "gfx/interface/icons/ig_icons/trade_unions.dds"
	layer = "revolution_dynamic_trade_unions"
	index = 8

	ideologies = {
		ideology_tbo_radical
		ideology_tbo_militant_socialist
		ideology_tbo_centralist
		ideology_tbo_isolationist
		ideology_tbo_institutionalist
	}

	enable = {
		THIS = c:FGT
		has_technology_researched = socialism
	}

	# Deprecated; use on_enable effect to assign traits instead
	traits = {
	}
	
	on_enable = {
		ig:ig_center_communist = {
			set_variable = special_ideology
			set_ig_trait = ig_trait:ig_trait_strikes
			set_ig_trait = ig_trait:ig_trait_industrial_organizers
			set_ig_trait = ig_trait:ig_trait_streamlining_production
		}
	}

	on_disable = {}
	on_character_ig_membership = {}

	pop_potential = {
		NOR = {
			is_pop_type = aristocrats
			is_pop_type = capitalists
		}
	}

	pop_weight = {
		value = 0

		add = {
			desc = "WORKERS"
			if = {
				limit = {
					OR = {
						pop_employment_building_group = bg_agriculture
						AND = {
							pop_employment_building_group = bg_mining
							strata = poor
						}
						AND = {
							pop_employment_building_group = bg_manufacturing
							strata = poor
						}
						AND = {
							pop_employment_building_group = bg_logging
							strata = poor
						}
				}
				value = 70
				if = {
					limit = {
						strata = middle
					}
					divide = 1.5
				}
			}
		}

		add = {
			desc = "ARMY"
			if = {
				limit = {
					OR = {
						pop_employment_building_group = bg_military
					}
				}
				value = 20
			}
		}
		
		add = {
			desc = "BUREAUCRATS"
			if = {
				limit = {
					OR = {
						pop_employment_building_group = bg_government
					}
				}
				value = 40
			}
		}

		add = {
			desc = "POP_MACHINISTS"
			if = {
				limit = {
					is_pop_type = machinists
				}
				value = 70
			}
		}
		
		add = {
			desc = "POP_PEASANTS"
			if = {
				limit = {
					is_pop_type = peasants
				}
				value = 100
			}
		}

		add = {
			desc = "POP_LABORERS"
			if = {
				limit = {
					is_pop_type = laborers
				}
				value = 100
			}
		}
		
		add = {
			desc = "SECURED_ARMY_SUPPORT"
			if = {
				limit = {
					pop_employment_building_group = bg_military
				}
				value = 40
			}
		}

		multiply = {
			desc = "LEADER_POPULARITY"
			scope:interest_group = {
				leader ?= {
					value = popularity
					multiply = 0.0025
					add = 1
					max = 1.25
					min = 0.75
				}
			}
		}
		
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = {
				desc = "IN_GOVERNMENT_ATTRACTION"
				value = 1
				add = scope:interest_group.modifier:interest_group_in_government_attraction_mult
				min = 0
			}
		}
		
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = yes
				}
			}
			multiply = { # Multiplied by 1.25x max
				desc = "POP_LOYALISTS"
				value = pop_loyalist_fraction
				divide = 4
				add = 1
			}
		}
		if = {
			limit = {
				scope:interest_group = {
					is_in_government = no
				}
			}
			multiply = { # Multiplied by 1.25x max
				desc = "POP_RADICALS"
				value = pop_radical_fraction
				divide = 4
				add = 1
			}
		}
	}

	monarch_weight = {
		value = 0.1 # this is just inherently unlikely
		# Monarch is not likely to adopt a marginal IG
		if = {
			limit = {
				is_marginal = yes
			}
			multiply = {
				value = 0.1
			}
		}
		# Monarch is more likely to adopt a Powerful IG
		if = {
			limit = {
				is_powerful = yes
			}
			multiply = {
				value = 2
			}
		}
		# More likely with Worker's Protections
		if = {
			limit = {
				owner = {
					has_law = law_type:law_worker_protections
				}
			}
			multiply = {
				value = 2
			}
		}
		# Impossible with No Workers Rights
		if = {
			limit = {
				owner = {
					has_law = law_type:law_no_workers_rights
				}
			}
			multiply = {
				value = 0
			}
		}
	}

	agitator_weight = {
		# Agitators more likely to come from populist IG's rather than elitist
		value = 1.0
		# Agitators will never be part of a marginalized IG
		if = {
			limit = {
				owner.ig:ig_trade_unions = {
					is_marginal = yes
				}
			}
			multiply = {
				value = 0
			}
		}
	}

	commander_weight = {
		value = 0.5
	}

	noble_chance = {
		value = 0.1
	}

	female_commander_chance = {
		value = 0.0
		
		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
			}
			add = {
				value = 0.2
			}
		}
		if = {
			limit = {
				owner = {
					exists = c:GDG
					THIS = c:GDG
				}
			}
			add = {
				value = 1.0
			}
		}
	}

	female_politician_chance = {
		value = 0.2

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}
			}
			add = {
				value = 0.3
			}
		}
		if = {
			limit = {
				owner = {
					exists = c:GDG
					THIS = c:GDG
				}
			}
			add = {
				value = 1.0
			}
		}
	}

	female_agitator_chance = {
		value = 0.2
		
		if = {
			limit = {
				owner = {
					exists = c:GDG
					THIS = c:GDG
				}
			}
			add = {
				value = 1.0
			}
		}
		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_own_property
				}

			}
			add = {
				value = 0.01
			}
		}	
		
		if = {
			limit = {
				owner = {
					has_law = law_type:law_women_in_the_workplace
				}

			}
			add = {
				value = 0.01
			}
		}

		if = {
			limit = {
				owner = {
					has_law = law_type:law_womens_suffrage
				}

			}
			add = {
				value = 0.01
			}
		}		
	}

	# The chance that a commander belonging to this IG takes over leadership when it changes
	# scope:character is the most popular commander in the IG
	commander_leader_chance = {
		value = 0.25
		if = {
			limit = {
				exists = scope:character
			}
			multiply = {
				value = 1
				add = scope:character.popularity
				min = 0
			}
		}
	}
}
